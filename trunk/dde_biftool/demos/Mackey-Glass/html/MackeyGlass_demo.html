load DDE-Biftool into path

clear
addpath('../../ddebiftool',...
    '../../ddebiftool_extra_psol',...
    '../../ddebiftool_utilities');

Enable vectorization

(disable for speed comparison)

x_vectorize=true;

Set user-defined functions

using gamma as constant and (beta,|n|,|tau|) as parameters

gamma=1.0;
beta_ind=1;
n_ind=2;
tau_ind=3;
if x_vectorize
    f=@(x,xtau,beta,n)beta*xtau./(1+xtau.^n)-gamma*x;
    funcs=set_funcs(...
        'sys_rhs',@(xx,p)f(xx(1,1,:),xx(1,2,:),p(1),p(2)),...
        'sys_tau',@()tau_ind,...
        'x_vectorized',true);
else
    f=@(x,xtau,beta,n)beta*xtau/(1+xtau^n)-gamma*x; %#ok<UNRCH>
    funcs=set_funcs(...
        'sys_rhs',@(xx,p)f(xx(1,1,:),xx(1,2,:),p(1),p(2)),...
        'sys_tau',@()tau_ind);
end

Initial parameters and state

beta0=2;
n0=10;
tau0=0;
x0=(beta0-1)^(1/n0);

Initialization of branch of non-trivial equilibria

contpar=tau_ind;

    'contpar',contpar,'max_step',[contpar,0.3],'max_bound',[contpar,10]);

Plot of period doubling bifurcation x profiles

pdsols={pd1sols,pd2sols};
floqpd={floqpd1,floqpd2};
get_par=@(i,k)arrayfun(@(x)x.parameter(i),pdsols{k});
figure(3)
clf;
subplot(3,2,[1,2]);
plot(get_par(beta_ind,1),get_par(tau_ind,1),'.-',...
    get_par(beta_ind,2),get_par(tau_ind,2),'.-');
xlabel('\beta');
ylabel('\tau');
title(sprintf(['1st and 2nd period doubling in Mackey-Glass eqn ',...
    'with n=%g, gamma=1'],n0));
grid on
for k=1:2
    subplot(3,2,2+k);
    hold on
    for i=1:length(pdsols{k})
        plot(pdsols{k}(i).mesh*pdsols{k}(i).period,pdsols{k}(i).profile,'-');
    end
    hold off
    box on
    grid on
    title(sprintf('PD%d: time profiles of period doubling',k));
    xlabel('t');
    ylabel('x');
    subplot(3,2,4+k);
    plot(1:length(pdsols{k}),floqpd{k}+1,'.-');
    grid on
    title(sprintf('PD%d: dist crit Floq mult from -1',k));
    ylabel('error');
    xlabel('point along branch');
end
